#Function to multiply two matrix
def input_matrix(row, col):
    matrix = []
    for i in range(row):
        row = []
        for j in range(col):
            elements = int(input(f"Enter the element of position {i+1}{j+1} : "))
            row.append(elements)
        matrix.append(row)
    return matrix

#Function to multiply 2 matrix
def mul_Matrix(matrix1, matrix2):
    row = len(matrix1)
    col = len(matrix2)
    result = []
    for i in range(row):
        row = []
        for j in range(col):
            row.append(matrix1[i][j]*matrix2[i][j])
            result.append(row)
    return result
if __name__ == "__main__":
    row = int(input("Enter the number of row : "))
    col = int(input("Enter the number of col : "))
    print("Enter elements for first matrix : ")
    matrix1 = input_matrix(row, col)
    print("Enter elements for second matrix : ")
    matrix2 = input_matrix(row, col)

    result_matrix = mul_Matrix(matrix1, matrix2)
    print("The resultant matrix after multiplication : ",)
    for row in result_matrix:

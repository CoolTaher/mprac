        <p id="Output"></p>
        <script type="text/javascript">
            function generate_fabonacci(){
                const numTerms = parseInt(document.getElementById('numterms').value);
                const Output = document.getElementById('Output');

                let first = 0, second = 1;
                
                let FabonacciSeries = [first];

                for (let i=1; i < numTerms; i++){
                    FabonacciSeries.push(second);
                    const next = first + second;
                    first = second;
                    second = next;
                }
                Output.textContent = "Fabonacci Series : " + FabonacciSeries.join(', ');
            }
        </script>

            function CheckVowels(){
                const str = document.getElementById('inputString').value.toLowerCase();
                const Output = document.getElementById('Output');

                if (str === ""){
                    Output.textContent = "Please enter a string";
                    return;
                }
                if (str.match([aeiouAEIOU])){
                    Output.textContent = "The string contains Vowel!!!";

                    Output.textContent = "The String Doesn't contain Vowel";

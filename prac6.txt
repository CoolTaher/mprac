def binarysearch(theValues,target):
    low = 0
    high = len(theValues)-1
    while low <= high:
        mid = (high+low) // 2
        if theValues[mid] == target:
            print("Number is found at poition ",mid + 1)
            break 
        elif target<theValues[mid]:
            high = mid - 1
        else:
            low = mid + 1
    else:
        print("Number is not available in the list.")

theValues = list(map(int,input("Enter the sorted list of numbers separated by spaces:").split()))
target = int(input("Enter the number to be searched:"))
binarysearch(theValues,target)

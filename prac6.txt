<html>
    <head>
        <title>Blur/change/focus/select/submit</title>
        <script src="jquery.js"></script>
    </head>
    <body>
        <h2>jquery Form Event</h2>
        <form id="myForm">
            <label for="name">Name:</label>
            <input type="text" id="name">  <br><br>

            <label id="city">City:</label>
            <select id="city">
                <option value="">Select City</option>
                <option value="Mumbai">Mumbai</option>
                <option value="Delhi">Delhi</option>
                <option value="Balgaluru">Balgaluru</option>
            </select> <br><br>

            <label id="message">Message</label>
            <textarea id="message" placeholder="Type here..."></textarea> <br><br>

            <input type="submit" value="Submit">
        </form>

        <p id="output"></p>

        <script>
            $(document).ready(function(){
                $("#name").focus(function(){
                    $("#output").text("Input Filed focused!");
                });
                $("#name").blur(function(){
                    $("#output").text("Input Filed lost focused(blur)");
                });
                $("#city").change(function(){
                    $("#output").text("DropDown value changed to: +$(this).val");
                });
                $("#message").select(function(){
                    $("#output").text("Text selected in textArea!");
                });
                $("#myForm").submit(function(){
                    event.preventDefault();
                    $("#output").text("Form Submitted");
                });
            });
        </script>
    </body>
</html>
